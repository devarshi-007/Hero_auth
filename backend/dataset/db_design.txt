Table credentials {
  id integer [primary key]
  username varchar [unique]
  password varchar
}

Table user_detail {
  id integer [primary key]
  name varchar
  birthdate timestamp
  deathdate timestamp
  originkey int
  organizationkey int
  retired bool
}

  fightid int [PK]
  hero_user_id int
  villain_user_id int
  winner_user_id int
}

  userid integer [primary key]
  weaponid int
}

Table weapon {
  weaponid integer [primary key]
  weapon varchar
  weaponclass rank
}

Table userpower {
  userid integer [primary key]
  powerkey varchar
}

Table power {
  powerkey integer [primary key]
  power varchar
}

Table userweekness {

  weeknesskey varchar
}

Table weekness {

  weekness varchar
}

Table origin {
  originkey integer [primary key]
  origin varchar
}

Table origanization {
  organizationkey integer [primary key]
  organization varchar
  type varchar
}


Table rank {
  userid integer [primary key]
  rankclass varchar
}

Table rankmap {
  userid int
  hero varchar
  villain varchar
}

Table mentor {
  userid int
  mentorid int
}

Table sidekick {
  userid int
  sidekick int
}
